//production Error
const sendErrorProd = (err,res)=>{
    //Operational Error, trusted errors send message to client
    // console.log("The value of isOperational in errorController is :- ",err.isOperational);
    if(err.isOperational){
        res.status(err.statuscode).json({
            status:err.status,
            message:err.message
        });
    }else{
        // console.error(`Error :- ${err}`);
        res.status(err.statuscode).json({
            status:err.status,
            message:"Production sensetive message..."
        });
    }
};

//Development Error
const sendErrorDev = (err,res)=>{
    res.status(err.statuscode).json({
            status:err.status,
            error:err,
            message:err.message,
            stack:err.stack
        });
};
